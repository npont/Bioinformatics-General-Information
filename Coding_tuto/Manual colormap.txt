# Manually create a matplotlib colormap gradient (violet in this case)

from matplotlib.colors import LinearSegmentedColormap
violet_colors = ["#f3e5f5", "#ce93d8", "#ab47bc", "#6a1b9a", "#4a148c"]
violet_cmap = LinearSegmentedColormap.from_list("violet_gradient", violet_colors)

# then replace cmap=violet_cmap in plot function

## where the color codes can be found at https://www.color-hex.com/color/f3e5f5
