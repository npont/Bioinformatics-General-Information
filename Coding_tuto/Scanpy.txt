### Raw and normalize - Best Practices ###
#### Method1 - using layers
# Save the raw counts as a layer
adata.layers["counts"] = adata.X.copy()
# Normalize
sc.pp.normalize_total(adata)
# Log transform
sc.pp.log1p(adata)
# Save the normalized version as a layer
adata["log_normalized"] = adata.X.copy()

#### Method2 - using raw
# OR save raw counts in raw
adata.raw = adata.copy() 
# it means now adata.raw.X contains the raw count matrix
... normalize ...
# after normalization the adata.X contains the normalized count matrix
# retrieve the raw counts with 
adata.raw.to_adata() that will then store into adata.X the raw version

###### Plot cells co-expressing marker ######
import numpy as np

# Indices of the genes:
idx_MYF5 = adata.var_names.get_loc('MYF5')
idx_MYOG = adata.var_names.get_loc('MYOG')

# Extract expression values; convert to dense if sparse
expr_MYF5 = adata.X[:, idx_MYF5].toarray().flatten() if hasattr(adata.X, "toarray") else adata.X[:, idx_MYF5]
expr_MYOG = adata.X[:, idx_MYOG].toarray().flatten() if hasattr(adata.X, "toarray") else adata.X[:, idx_MYOG]

# Find cells where both markers are expressed (expression > 0)
coexpress_mask = (expr_MYF5 > 0) & (expr_MYOG > 0)

# Subset adata to these cells:
adata_coexpress = adata[coexpress_mask].copy()

adata.obs['MYF5_MYOG_coexpress'] = coexpress_mask

sc.pl.umap(adata, color='MYF5_MYOG_coexpress')
